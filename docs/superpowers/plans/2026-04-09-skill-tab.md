# 스킬 탭 (Plan 1: 만들기 + 목록) Implementation Plan v2

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 사이드바에 "스킬" 탭을 추가해, 사용자가 Claude Code skill 형태의 재사용 가능한 워크플로우를 대화형으로 만들고(1번째 패널) 목록에서 조회할 수 있게(2번째 패널) 한다. 스케줄링 및 카드 클릭 실행 패널은 후속 Plan 2에서 다룬다.

**Architecture:** 탭 셸은 기존 `mode-company-card.js`의 `_bootMode` 스위치에 `case 'skill'`을 추가하는 패턴을 따른다. 백엔드는 `src/skill_builder/` 신규 모듈에서 ① 한국어 진입 UX, ② 기존 스킬/마켓플레이스 검색 + `install_counts_cache.json` 필터, ③ 공식 `skill-creator` 스킬을 `ClaudeCodeBridge.raw_query`의 **세션 resume 루프**로 다중 턴 호출하는 브리지, ④ `data/skills/registry.json` 메타데이터 저장소를 담당한다. 생성된 스킬은 `~/.claude/skills/skill-tab-{slug}/` 프리픽스 경로에 저장되어 앱이 자기 스킬을 식별할 수 있게 한다.

**Tech Stack:** Python 3.11 + FastAPI(WebSocket), 기존 `ClaudeCodeBridge`(`src/utils/claude_code.py`), 바닐라 JS(Module 패턴), pytest.

```python
"""스킬 레지스트리 — data/skills/registry.json 단일 파일 저장소."""

from __future__ import annotations

import json
import tempfile
from dataclasses import asdict, dataclass
from pathlib import Path
from typing import List


@dataclass
class SkillRecord:
    slug: str
    name: str
    skill_path: str
    required_mcps: List[str]
    source: str  # "created" | "imported"
    created_at: str  # ISO 8601


class SkillRegistry:
    def __init__(self, path: Path) -> None:
        self.path = Path(path)

    def list_all(self) -> List[SkillRecord]:
        if not self.path.exists():
            return []
        raw = json.loads(self.path.read_text() or "[]")
        return [SkillRecord(**r) for r in raw]

    def add(self, record: SkillRecord) -> None:
        current = self.list_all()
        if any(r.slug == record.slug for r in current):
            raise ValueError(f"slug '{record.slug}' 이미 존재합니다")
        current.append(record)
        self._atomic_write([asdict(r) for r in current])

    def _atomic_write(self, data: list) -> None:
        """임시 파일 + rename으로 원자적 쓰기 (부분 쓰기 방지)."""
        self.path.parent.mkdir(parents=True, exist_ok=True)
        with tempfile.NamedTemporaryFile(
            mode="w", dir=str(self.path.parent), delete=False,
            suffix=".tmp", encoding="utf-8",
        ) as tmp:
            json.dump(data, tmp, ensure_ascii=False, indent=2)
            tmp_path = Path(tmp.name)
        tmp_path.replace(self.path)
```

`src/skill_builder/skill_search.py`:
```python
"""기존 Claude Code 스킬을 검색하여 "이미 만들어진 것" 여부를 판단.

검색 소스:
  1. ~/.claude/skills/          — 사용자 머신에 설치된 스킬
  2. ~/.claude/plugins/install-counts-cache.json — 정량 필터 기준

필터 순서:
  1) skill-tab- 프리픽스 제외 (이미 앱이 만든 것)
  2) trusted_marketplaces.json의 min_unique_installs 임계값 필터
  3) description·name에 대한 단순 키워드 매칭
"""

from __future__ import annotations

import json
import os
import re
from dataclasses import dataclass
from pathlib import Path
from typing import List, Optional

_CONFIG_PATH = Path(__file__).parent / "trusted_marketplaces.json"


@dataclass
class SkillCandidate:
    slug: str
    name: str
    description: str
    source_marketplace: Optional[str]
    unique_installs: int
    skill_md_path: Path


class SkillSearchIndex:
    def __init__(
        self,
        candidates: List[SkillCandidate],
        min_installs: int,
    ) -> None:
        self._candidates = candidates
        self._min_installs = min_installs

    @classmethod
    def load(cls) -> "SkillSearchIndex":
        cfg = json.loads(_CONFIG_PATH.read_text())
        min_installs = cfg.get("min_unique_installs", 1000)

        home = Path(os.environ.get("HOME", "~")).expanduser()
        skills_dir = home / ".claude" / "skills"
        counts_path = home / ".claude" / "plugins" / "install-counts-cache.json"

        install_counts = cls._load_install_counts(counts_path)

        candidates: List[SkillCandidate] = []
        if skills_dir.exists():
            for skill_dir in sorted(skills_dir.iterdir()):
                if not skill_dir.is_dir():
                    continue
                if skill_dir.name.startswith("skill-tab-"):
                    continue  # 앱이 만든 스킬은 제외
                skill_md = skill_dir / "SKILL.md"
                if not skill_md.exists():
                    continue
                name, desc = cls._parse_frontmatter(skill_md)
                installs = install_counts.get(skill_dir.name, 0)
                candidates.append(SkillCandidate(
                    slug=skill_dir.name,
                    name=name,
                    description=desc,
                    source_marketplace=None,
                    unique_installs=installs,
                    skill_md_path=skill_md,
                ))
        return cls(candidates=candidates, min_installs=min_installs)

    def search(self, query: str) -> List[SkillCandidate]:
        terms = [t.lower() for t in re.findall(r"\w+", query) if len(t) > 2]
        if not terms:
            return []
        matches: list[tuple[int, SkillCandidate]] = []
        for c in self._candidates:
            if c.unique_installs < self._min_installs:
                continue
            haystack = f"{c.name} {c.description}".lower()
            score = sum(1 for t in terms if t in haystack)
            if score > 0:
                matches.append((score, c))
        matches.sort(key=lambda x: (-x[0], -x[1].unique_installs))
        return [c for _, c in matches]

    @staticmethod
    def _load_install_counts(path: Path) -> dict:
        if not path.exists():
            return {}
        raw = json.loads(path.read_text())
        result = {}
        for entry in raw.get("counts", []):
            plugin = entry.get("plugin", "")
            # "name@marketplace" 형태에서 name만 추출
            slug = plugin.split("@", 1)[0]
            result[slug] = entry.get("unique_installs", 0)
        return result

    @staticmethod
    def _parse_frontmatter(path: Path) -> tuple[str, str]:
        text = path.read_text()
        if not text.startswith("---"):
            return path.parent.name, ""
        parts = text.split("---", 2)
        if len(parts) < 3:
            return path.parent.name, ""
        fm = parts[1]
        name = re.search(r"^name:\s*(.+)$", fm, re.MULTILINE)
        desc = re.search(r"^description:\s*(.+)$", fm, re.MULTILINE)
        return (
            name.group(1).strip() if name else path.parent.name,
            desc.group(1).strip() if desc else "",
        )
```

```python
## Completion Signal

When you have finished creating the skill and are satisfied with it:
1. Verify `~/.claude/skills/skill-tab-{slug}/SKILL.md` exists
2. End your final message with the exact token: [SKILL_COMPLETE]

      6) 완료 시 [SKILL_COMPLETE] 토큰 emit (여기서)
    """
    base = build_skill_creator_directive(user_description)
    base = base.replace("{slug}", slug)
```

```python
"""스킬 만들기 세션 오케스트레이터.

핵심 책임 (순서대로):
  1) "start" 메시지 수신 → 한국어 인사
  2) 기존 스킬 검색 → 결과 emit
  3) 사용자 선택 대기 ("new" 또는 "import:<slug>")
  4) "new"면 skill-creator 세션 resume 루프 진입 (Step 4.2)
  5) 완료 후 검증 + 레지스트리 등록

세션 resume 루프 (Step 4.2에서 구현):
  - raw_query(session_id=...)로 첫 턴 시작
  - Claude 응답을 사용자에게 전달
  - 사용자 답변(type="user_message") 수신
  - raw_query(resume=session_id, user_message=답변)으로 다음 턴
  - [SKILL_COMPLETE] 토큰이 응답에 나타날 때까지 반복
  - 최대 15라운드 제한
"""

from __future__ import annotations

import re
import uuid
from dataclasses import asdict
from datetime import datetime, timezone
from pathlib import Path
from typing import TYPE_CHECKING

from src.skill_builder.clarification_questions import build_entry_greeting
from src.skill_builder.skill_search import SkillSearchIndex

if TYPE_CHECKING:
    from fastapi import WebSocket


MAX_INTERVIEW_ROUNDS = 15
COMPLETION_TOKEN = "[SKILL_COMPLETE]"


def _slugify(text: str) -> str:
    """한국어/영어 혼합 입력을 ASCII slug로 변환.
    한글만 있으면 timestamp fallback (skill- prefix로 네임스페이스 명확화)."""
    ascii_only = re.sub(r"[^a-zA-Z0-9]+", "-", text).strip("-").lower()
    if ascii_only:
        return ascii_only[:40]
    # 한국어 전용 입력 fallback — 충돌 방지를 위해 timestamp + short uuid
    ts = int(datetime.now(timezone.utc).timestamp())
    short = uuid.uuid4().hex[:6]
    return f"skill-{ts}-{short}"


async def run_skill_builder_session(ws: "WebSocket") -> None:
    """단일 WebSocket 세션을 종료까지 처리."""
    # 1) start 메시지 대기
    msg = await ws.receive_json()
    if msg.get("type") != "start":
        await ws.send_json({
            "type": "error",
            "data": {"message": "첫 메시지는 'start' 타입이어야 합니다"},
        })
        return

    description = (msg.get("data") or {}).get("description", "").strip()
    if not description:
        await ws.send_json({
            "type": "error",
            "data": {"message": "스킬 설명을 입력해주세요"},
        })
        return

    # 2) 한국어 인사
    await ws.send_json({
        "type": "greeting",
        "data": {"text": build_entry_greeting(description)},
    })

    # 3) 기존 스킬 검색 (실패는 치명적이지 않음)
    try:
        index = SkillSearchIndex.load()
        candidates = index.search(description)
    except Exception:
        candidates = []

    await ws.send_json({
        "type": "search_results",
        "data": {
            "candidates": [
                {
                    "slug": c.slug,
                    "name": c.name,
                    "description": c.description,
                    "unique_installs": c.unique_installs,
                }
                for c in candidates[:5]
            ],
        },
    })

    # 4) 사용자 선택 대기
    choice_msg = await ws.receive_json()
    if choice_msg.get("type") == "cancel":
        return
    if choice_msg.get("type") != "choice":
        await ws.send_json({
            "type": "error",
            "data": {"message": "선택 메시지(choice)를 기다리고 있었습니다"},
        })
        return

    choice = (choice_msg.get("data") or {}).get("choice", "new")
    if choice != "new":
        # "import:<slug>" 처리는 후속 버전
        await ws.send_json({
            "type": "error",
            "data": {"message": "기존 스킬 가져오기는 다음 버전에서 지원 예정"},
        })
        return

    # 5) 세션 resume 루프 (Step 4.2에서 추가)
    await _run_skill_creator_loop(ws, description)
```

```python
async def _run_skill_creator_loop(
    ws: "WebSocket",
    description: str,
) -> None:
    """skill-creator를 세션 resume 방식으로 다중 턴 호출.

    핵심 로직:
      - session_id를 한 번 생성
      - 첫 턴: raw_query(session_id=..., user_message=description)
      - 이후 N턴: 사용자가 ws로 user_message 보낼 때마다
                 raw_query(resume=session_id, user_message=답변)
      - 응답에 [SKILL_COMPLETE] 등장 또는 MAX_INTERVIEW_ROUNDS 초과 시 종료
    """
    from src.skill_builder.skill_creator_bridge import build_handoff_system_prompt
    from src.skill_builder.registry import SkillRegistry, SkillRecord
    from src.utils.claude_code import ClaudeCodeBridge
    from src.config.settings import get_settings

    slug = _slugify(description)
    system_prompt = build_handoff_system_prompt(
        user_description=description, slug=slug,
    )

    bridge = ClaudeCodeBridge()
    settings = get_settings()
    session_id = str(uuid.uuid4())

    # 첫 턴: 세션 시작
    # allowed_tools는 모두 builtin이므로 _all_builtin()=True → skip_mcp=True
    # → 자동으로 /tmp에서 실행되어 우리 CLAUDE.md 누출 방지됨
    allowed = ["Read", "Write", "Edit", "Bash", "Glob", "Grep"]

    try:
        response = await bridge.raw_query(
            system_prompt=system_prompt,
            user_message=description,
            model="sonnet",
            allowed_tools=allowed,
            max_turns=30,
            timeout=settings.planning_timeout,
            session_id=session_id,
        )
    except Exception as e:
        await ws.send_json({
            "type": "error",
            "data": {"message": f"세션 시작 실패: {e}"},
        })
        return

    await ws.send_json({
        "type": "assistant_message",
        "data": {"text": response},
    })

    # Resume 루프
    for round_num in range(MAX_INTERVIEW_ROUNDS):
        if COMPLETION_TOKEN in response:
            break

        # 사용자 답변 대기
        reply_msg = await ws.receive_json()
        if reply_msg.get("type") == "cancel":
            return
        if reply_msg.get("type") != "user_message":
            await ws.send_json({
                "type": "error",
                "data": {"message": "user_message 타입을 기다리고 있었습니다"},
            })
            return

        user_text = (reply_msg.get("data") or {}).get("text", "").strip()
        if not user_text:
            continue

        try:
            response = await bridge.raw_query(
                system_prompt=system_prompt,
                user_message=user_text,
                model="sonnet",
                allowed_tools=allowed,
                max_turns=30,
                timeout=settings.planning_timeout,
                resume=session_id,
            )
        except Exception as e:
            await ws.send_json({
                "type": "error",
                "data": {"message": f"세션 resume 실패 (라운드 {round_num+1}): {e}"},
            })
            return

        await ws.send_json({
            "type": "assistant_message",
            "data": {"text": response},
        })
    else:
        # for-else: break 없이 루프 종료 = 최대 라운드 초과
        await ws.send_json({
            "type": "error",
            "data": {
                "message": f"최대 대화 라운드({MAX_INTERVIEW_ROUNDS}) 초과. "
                           "스킬이 완성되지 못했습니다.",
            },
        })
        return

    # 6) 검증: 스킬 파일 존재 + subagent는 존재하지 않음
    home = Path.home()
    skill_path = home / ".claude" / "skills" / f"skill-tab-{slug}"
    skill_md = skill_path / "SKILL.md"
    subagent_path = home / ".claude" / "agents" / f"skill-tab-{slug}.md"

    if subagent_path.exists():
        await ws.send_json({
            "type": "error",
            "data": {
                "message": "skill-creator가 스킬 대신 subagent를 만들었습니다. "
                           "핸드오프 프롬프트 확인이 필요합니다.",
            },
        })
        return

    if not skill_md.exists():
        await ws.send_json({
            "type": "error",
            "data": {
                "message": "완료 토큰은 받았지만 SKILL.md 파일이 없습니다. "
                           f"예상 경로: {skill_md}",
            },
        })
        return

    # 7) 레지스트리 등록
    reg = SkillRegistry(path=Path("data/skills/registry.json"))

    # skill_metadata.json에서 required_mcps 읽기 (없으면 빈 리스트)
    required_mcps: list[str] = []
    meta_path = skill_path / "skill_metadata.json"
    if meta_path.exists():
        try:
            import json as _json
            meta = _json.loads(meta_path.read_text())
            required_mcps = meta.get("required_mcps", []) or []
        except Exception:
            pass  # 파싱 실패는 무시 — 빈 리스트로 진행

    record = SkillRecord(
        slug=slug,
        name=description[:50],
        skill_path=str(skill_path),
        required_mcps=required_mcps,
        source="created",
        created_at=datetime.now(timezone.utc).isoformat(),
    )
    try:
        reg.add(record)
    except ValueError:
        # 동일 slug 충돌 (드문 경우) — timestamp suffix 재시도
        record.slug = f"{slug}-{int(datetime.now(timezone.utc).timestamp())}"
        reg.add(record)

    await ws.send_json({
        "type": "created",
        "data": {"slug": record.slug, "skill_path": record.skill_path},
    })
```

```javascript
 * WebSocket /ws/skill-builder 연결 (create 패널)
 * REST /api/skill-builder/list (list 패널)
 *
 * Create 패널은 skill-creator와의 다중 턴 대화를 지원:
 *   - "만들기 시작" 후 WebSocket open
 *   - 서버가 search_results 보내면 자동으로 choice:new 응답
 *   - assistant_message가 올 때마다 답변 입력란 활성화
 *   - 사용자 답변은 user_message 타입으로 전송
 */
var SkillManager = (function () {
  'use strict';

  var _container = null;
  var _activePanel = 'create';  // 'create' | 'list'
  var _ws = null;
  var _replyInput = null;
  var _replyBtn = null;

  function mountInShell(container) {
    _container = container;
    _render();
  }

  function _render() {
    if (!_container) return;
    while (_container.firstChild) _container.removeChild(_container.firstChild);

    var switcher = document.createElement('div');
    switcher.className = 'skill-panel-switcher';
    [['create', '스킬 만들기'], ['list', '내 스킬']].forEach(function (pair) {
      var btn = document.createElement('button');
      btn.textContent = pair[1];
      btn.className = (pair[0] === _activePanel) ? 'active' : '';
      btn.onclick = function () { _activePanel = pair[0]; _render(); };
      switcher.appendChild(btn);
    });
    _container.appendChild(switcher);

    var panel = document.createElement('div');
    panel.className = 'skill-panel';
    _container.appendChild(panel);

    if (_activePanel === 'create') {
      _renderCreatePanel(panel);
    } else {
      _renderListPanel(panel);
    }
  }

  function _renderCreatePanel(root) {
    var title = document.createElement('h3');
    title.textContent = '새 스킬 만들기';
    root.appendChild(title);

    var sub = document.createElement('p');
    sub.className = 'subtitle';
    sub.textContent = 'Claude가 따를 재사용 가능한 워크플로우를 한 번 만들어두고 언제든 불러쓰세요.';
    root.appendChild(sub);

    var input = document.createElement('input');
    input.className = 'skill-create-input';
    input.placeholder = '예: 매일 아침 경쟁사 뉴스 3개를 요약해줘';
    root.appendChild(input);

    var log = document.createElement('div');
    log.className = 'skill-chat-log';
    log.textContent = '[준비됨] 설명을 입력하고 "만들기 시작" 버튼을 눌러주세요.';
    root.appendChild(log);

    var startBtn = document.createElement('button');
    startBtn.className = 'skill-create-btn';
    startBtn.textContent = '만들기 시작';
    startBtn.onclick = function () {
      var desc = input.value.trim();
      if (!desc) return;
      input.disabled = true;
      startBtn.disabled = true;
      _startSession(desc, log);
    };
    root.appendChild(startBtn);

    // 답변 입력 row (처음엔 숨김)
    var replyRow = document.createElement('div');
    replyRow.className = 'skill-reply-row';
    replyRow.style.display = 'none';
    _replyInput = document.createElement('input');
    _replyInput.className = 'skill-create-input';
    _replyInput.placeholder = '답변을 입력하세요...';
    _replyBtn = document.createElement('button');
    _replyBtn.className = 'skill-reply-btn';
    _replyBtn.textContent = '보내기';
    _replyBtn.onclick = function () { _sendReply(log); };
    _replyInput.addEventListener('keydown', function (e) {
      if (e.key === 'Enter') _sendReply(log);
    });
    replyRow.appendChild(_replyInput);
    replyRow.appendChild(_replyBtn);
    root.appendChild(replyRow);
    _replyInput._row = replyRow;
  }

  function _sendReply(logEl) {
    if (!_ws || _ws.readyState !== 1) return;
    var text = _replyInput.value.trim();
    if (!text) return;
    _appendLog(logEl, '> ' + text);
    _ws.send(JSON.stringify({ type: 'user_message', data: { text: text } }));
    _replyInput.value = '';
    _replyInput.disabled = true;
    _replyBtn.disabled = true;
  }

  function _appendLog(logEl, text) {
    logEl.textContent += '\n' + text;
    logEl.scrollTop = logEl.scrollHeight;
  }

  function _enableReply() {
    if (_replyInput && _replyInput._row) {
      _replyInput._row.style.display = '';
      _replyInput.disabled = false;
      _replyBtn.disabled = false;
      _replyInput.focus();
    }
  }

  function _renderListPanel(root) {
    var title = document.createElement('h3');
    title.textContent = '내 스킬';
    root.appendChild(title);

    var sub = document.createElement('p');
    sub.className = 'subtitle';
    sub.textContent = '지금까지 만든 스킬들입니다. 카드 클릭 실행은 다음 버전에서 지원 예정.';
    root.appendChild(sub);

    var grid = document.createElement('div');
    grid.className = 'skill-list-grid';
    root.appendChild(grid);

    fetch('/api/skill-builder/list')
      .then(function (r) { return r.json(); })
      .then(function (data) {
        var skills = (data && data.skills) || [];
        if (!skills.length) {
          var empty = document.createElement('p');
          empty.className = 'skill-empty';
          empty.textContent = '아직 만든 스킬이 없어요. "스킬 만들기" 탭에서 시작하세요.';
          grid.appendChild(empty);
          return;
        }
        skills.forEach(function (s) {
          var card = document.createElement('div');
          card.className = 'skill-card';
          var h = document.createElement('h4');
          h.textContent = s.name || s.slug;
          card.appendChild(h);
          var meta = document.createElement('div');
          meta.className = 'meta';
          meta.textContent = (s.source === 'created' ? '직접 생성' : '가져옴')
            + ' · ' + (s.created_at || '').slice(0, 10);
          card.appendChild(meta);
          grid.appendChild(card);
        });
      })
      .catch(function (e) {
        var err = document.createElement('p');
        err.className = 'skill-empty';
        err.textContent = '목록을 불러오지 못했어요: ' + e.message;
        grid.appendChild(err);
      });
  }

  function _startSession(description, logEl) {
    logEl.textContent = '';
    _appendLog(logEl, '[연결됨] 세션 시작...');

    if (_ws) { try { _ws.close(); } catch (e) {} }
    var proto = location.protocol === 'https:' ? 'wss://' : 'ws://';
    _ws = new WebSocket(proto + location.host + '/ws/skill-builder');

    _ws.onopen = function () {
      _ws.send(JSON.stringify({ type: 'start', data: { description: description } }));
    };

    _ws.onmessage = function (ev) {
      var msg;
      try { msg = JSON.parse(ev.data); } catch (e) { return; }
      if (msg.type === 'greeting') {
        _appendLog(logEl, msg.data.text);
      } else if (msg.type === 'search_results') {
        var cs = msg.data.candidates || [];
        if (cs.length) {
          _appendLog(logEl, '\n[기존 스킬 ' + cs.length + '개 발견]');
          cs.forEach(function (c) {
            _appendLog(logEl, '  • ' + c.name + ' (설치 ' + c.unique_installs + ')');
          });
          _appendLog(logEl, '\n[새로 만들기로 진행합니다]');
        } else {
          _appendLog(logEl, '\n[기존 스킬 없음 — 새로 만들기로 진행]');
        }
        // 현재 버전: 항상 새로 만들기
        _ws.send(JSON.stringify({ type: 'choice', data: { choice: 'new' } }));
      } else if (msg.type === 'assistant_message') {
        _appendLog(logEl, '\n' + ((msg.data && msg.data.text) || ''));
        _enableReply();
      } else if (msg.type === 'created') {
        _appendLog(logEl, '\n[✅ 완료] 스킬이 저장되었습니다: ' + msg.data.slug);
        if (_replyInput && _replyInput._row) _replyInput._row.style.display = 'none';
      } else if (msg.type === 'error') {
        _appendLog(logEl, '\n[❌ 오류] ' + (msg.data && msg.data.message));
        if (_replyInput && _replyInput._row) _replyInput._row.style.display = 'none';
      }
    };

    _ws.onerror = function () { _appendLog(logEl, '\n[오류] WebSocket 연결 실패'); };
    _ws.onclose = function () { _appendLog(logEl, '\n[세션 종료]'); };
  }

  return { mountInShell: mountInShell };
})();
```

## 주의사항 및 알려진 제약

1. **skill-creator 다중 턴의 테스트 가능성**: 외부 프로세스 + LLM 호출이라 진정한 자동 단위 테스트는 불가능. Task 2에서는 프롬프트 빌드 로직만 테스트. 실제 루프 검증은 Task 6의 수동 플로우에 의존.

2. **완료 토큰 누락 리스크**: skill-creator가 handoff 지시를 무시하고 `[SKILL_COMPLETE]`를 안 찍으면 최대 라운드까지 헛돌다 실패. handoff 프롬프트에 강하게 명시했지만 100% 보장은 아님. 실패 시 에러 메시지는 명확히 표시됨.

3. **세션 상태 디스크 저장**: Claude Code CLI는 `session_id`로 시작한 세션을 로컬 디스크에 저장(일반적으로 `~/.claude/projects/`). 우리 runner는 이 저장 자체에 관여하지 않음 — CLI가 알아서 함. 중단된 세션은 고아로 남을 수 있음.

4. **검색 품질**: Task 2의 키워드 매칭은 단순(lowercase substring). 임베딩 기반 업그레이드는 후속 과제. "정확히 매치 안 되면 새로 만들기"가 fallback이라 큰 문제는 아님.

5. **`install_counts_cache.json` 신선도**: Claude Code가 주기적으로 갱신. 캐시가 없거나 비어있으면 모든 스킬이 `unique_installs=0`으로 필터링되어 검색 결과 없음 → "기존 스킬 없음 → 생성" 플로우로 자연스럽게 진행.

6. **동시성**: 한 클라이언트가 여러 세션을 동시에 열면 각 WebSocket이 독립. 같은 slug로 동시 생성 시 `SkillRegistry.add()`가 race condition에 노출되지만 atomic write로 파일 손상은 방지되며 slug 충돌은 Task 4.2의 timestamp suffix fallback으로 부분 해결.

7. **CLAUDE.md 누출 방지**: `allowed_tools` = `["Read","Write","Edit","Bash","Glob","Grep"]`가 모두 `_BUILTIN_TOOLS`에 있어 `_all_builtin()=True` → `skip_mcp=True` → subprocess가 `/tmp`에서 실행 (`claude_code.py:1180-1182`). 따라서 skill-creator는 우리 프로젝트의 CLAUDE.md를 보지 못함. 별도 `cwd` 파라미터 추가 불필요.

8. **용어 일관성**: 코드·프롬프트·UI 모두 "skill"/"스킬"로 통일. "agent"/"에이전트"는 사용하지 않음 (기존 코드베이스의 `src/agents/` CEO/Leader/Worker와 분리하기 위함).
